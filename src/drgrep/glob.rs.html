1//! # Enhanced Glob Pattern Matcher
2//! 
3//! A comprehensive glob pattern matcher implementation in Rust.
4//! This module provides functionality to match strings against glob patterns
5//! with support for:
6//! - `*` (matches any sequence of characters)
7//! - `?` (matches any single character)
8//! - `[abc]` (matches any character in the set)
9//! - `[!abc]` (matches any character not in the set)
10//! - `{a,b,c}` (matches any of the comma-separated patterns)
11//! - File system traversal to find matching files
12
13use std::collections::HashSet;
14use std::fs;
15use std::path::{Path, PathBuf};
16
17/// Represents a compiled glob pattern for efficient matching.
18#[derive(Debug, Clone)]
19pub struct GlobPattern {
20    pattern: String,
21    components: Vec<Component>,
22    expanded_patterns: Vec<String>, // For alternatives
23}
24
25/// Components that make up a glob pattern.
26#[derive(Debug, Clone)]
27enum Component {
28    /// Matches a literal string
29    Literal(String),
30    /// Matches any single character
31    SingleWildcard,
32    /// Matches any sequence of characters (including empty)
33    MultiWildcard,
34    /// Matches any character in the set
35    CharacterClass {
36        chars: HashSet<char>,
37        negated: bool,
38    },
39    // Matches any of the comma-separated patterns
40    // Alternatives(Vec<String>),
41}
42
43impl GlobPattern {
44    /// Creates a new `GlobPattern` instance from a pattern string.
45    ///
46    /// # Arguments
47    ///
48    /// * `pattern` - A string slice containing the glob pattern
49    ///
50    /// # Examples
51    ///
52    /// ```
53    /// use drgrep::glob::GlobPattern;
54    ///
55    /// let pattern = GlobPattern::new("*.rs");
56    /// assert!(pattern.matches("file.rs"));
57    /// assert!(!pattern.matches("file.txt"));
58    /// ```
59    pub fn new(pattern: &str) -> Self {
60        let pattern_string = pattern.to_string();
61        let (components, expanded_patterns) = Self::parse(&pattern_string);
62        
63        GlobPattern {
64            pattern: pattern_string,
65            components,
66            expanded_patterns,
67        }
68    }
69
70    /// Parse a glob pattern string into components and expanded alternative patterns.
71    fn parse(pattern: &str) -> (Vec<Component>, Vec<String>) {
72        // First, check if we have any braces that indicate alternatives
73        if pattern.contains('{') && pattern.contains('}') {
74            let expanded = Self::expand_alternatives(pattern);
75            return (Vec::new(), expanded);
76        }
77        
78        // If no alternatives, proceed with normal parsing
79        let mut components = Vec::new();
80        let mut current_literal = String::new();
81        let chars: Vec<char> = pattern.chars().collect();
82        let mut i = 0;
83        
84        while i < chars.len() {
85            match chars[i] {
86                '*' => {
87                    if !current_literal.is_empty() {
88                        components.push(Component::Literal(current_literal));
89                        current_literal = String::new();
90                    }
91                    components.push(Component::MultiWildcard);
92                    i += 1;
93                },
94                '?' => {
95                    if !current_literal.is_empty() {
96                        components.push(Component::Literal(current_literal));
97                        current_literal = String::new();
98                    }
99                    components.push(Component::SingleWildcard);
100                    i += 1;
101                },
102                '[' => {
103                    if !current_literal.is_empty() {
104                        components.push(Component::Literal(current_literal));
105                        current_literal = String::new();
106                    }
107                    
108                    let (char_class, new_pos) = Self::parse_character_class(&chars, i + 1);
109                    components.push(char_class);
110                    i = new_pos;
111                },
112                '\\' => {
113                    // Escape the next character
114                    if i + 1 < chars.len() {
115                        current_literal.push(chars[i + 1]);
116                        i += 2;
117                    } else {
118                        current_literal.push('\\');
119                        i += 1;
120                    }
121                },
122                _ => {
123                    current_literal.push(chars[i]);
124                    i += 1;
125                }
126            }
127        }
128        
129        if !current_literal.is_empty() {
130            components.push(Component::Literal(current_literal));
131        }
132        
133        (components, Vec::new())
134    }

136    /// Expand alternatives in a pattern like {a,b} to multiple patterns
137    fn expand_alternatives(pattern: &str) -> Vec<String> {
138        let mut result = Vec::new();
139        let mut current = String::new();
140        
141        // First, identify top-level alternative groups (not nested)
142        let mut i = 0;
143        let chars: Vec<char> = pattern.chars().collect();
144        
145        while i < chars.len() {
146            match chars[i] {
147                '{' => {
148                    // Find the matching closing brace
149                    let start_pos = i;
150                    let mut brace_count = 1;
151                    let mut end_pos = start_pos;
152                    
153                    for j in (start_pos + 1)..chars.len() {
154                        match chars[j] {
155                            '{' => brace_count += 1,
156                            '}' => {
157                                brace_count -= 1;
158                                if brace_count == 0 {
159                                    end_pos = j;
160                                    break;
161                                }
162                            },
163                            _ => {}
164                        }
165                    }
166                    
167                    if brace_count != 0 || end_pos == start_pos {
168                        // Malformed pattern, treat as literal
169                        current.push('{');
170                        i += 1;
171                        continue;
172                    }
173                    
174                    // Extract the options within braces
175                    let options_str = &pattern[start_pos + 1..end_pos];
176                    
177                    // Split by commas, but respect nested braces
178                    let options = Self::split_respecting_braces(options_str);
179                    
180                    // Recursive expansion for each option
181                    let prefix = current.clone();
182                    let suffix = &pattern[end_pos + 1..];
183                    
184                    // If this is the first alternative, initialize result
185                    if result.is_empty() {
186                        for option in options {
187                            // Recursively expand any alternatives in this option
188                            let expanded_option = Self::expand_alternatives(&format!("{}{}{}", prefix, option, suffix));
189                            result.extend(expanded_option);
190                        }
191                    } else {
192                        // For subsequent alternatives, apply each option to every existing result
193                        let mut new_results = Vec::new();
194                        
195                        for existing in &result {
196                            for option in &options {
197                                // Recursively expand
198                                let expanded_option = Self::expand_alternatives(&format!("{}{}", option, suffix));
199                                for exp in expanded_option {
200                                    new_results.push(format!("{}{}", existing.clone(), exp));
201                                }
202                            }
203                        }
204                        
205                        if !new_results.is_empty() {
206                            result = new_results;
207                        }
208                    }
209                    
210                    
211                    // Since we've processed this alternative completely, we can return
212                    return result;
213                },
214                '\\' => {
215                    // Escape the next character
216                    if i + 1 < chars.len() {
217                        current.push(chars[i + 1]);
218                        i += 2;
219                    } else {
220                        current.push('\\');
221                        i += 1;
222                    }
223                },
224                _ => {
225                    current.push(chars[i]);
226                    i += 1;
227                }
228            }
229        }
230        
231        // If we get here with no expansion, add the literal pattern
232        if result.is_empty() {
233            result.push(current);
234        }
235        
236        result
237    }

239    /// Split a string by commas, but respect nested braces
240    fn split_respecting_braces(s: &str) -> Vec<String> {
241        let mut result = Vec::new();
242        let mut current = String::new();
243        let mut brace_count = 0;
244        
245        for c in s.chars() {
246            match c {
247                '{' => {
248                    brace_count += 1;
249                    current.push(c);
250                },
251                '}' => {
252                    brace_count -= 1;
253                    current.push(c);
254                },
255                ',' if brace_count == 0 => {
256                    result.push(current);
257                    current = String::new();
258                },
259                _ => current.push(c)
260            }
261        }
262        
263        if !current.is_empty() {
264            result.push(current);
265        }
266        
267        result
268    }

270    /// Parse a character class like [abc] or [!abc]
271    fn parse_character_class(chars: &[char], start: usize) -> (Component, usize) {
272        let mut i = start;
273        let mut char_set = HashSet::new();
274        let mut negated = false;
275        
276        // Check if the class is negated
277        if i < chars.len() && chars[i] == '!' {
278            negated = true;
279            i += 1;
280        }
281        
282        // Parse the class content
283        while i < chars.len() && chars[i] != ']' {
284            if i + 2 < chars.len() && chars[i + 1] == '-' {
285                // Character range like a-z
286                let start_char = chars[i];
287                let end_char = chars[i + 2];
288                
289                for c in start_char..=end_char {
290                    char_set.insert(c);
291                }
292                
293                i += 3;
294            } else {
295                char_set.insert(chars[i]);
296                i += 1;
297            }
298        }
299        
300        // Skip closing bracket
301        if i < chars.len() {
302            i += 1;
303        }
304        
305        (Component::CharacterClass { chars: char_set, negated }, i)
306    }

313    /// Checks if a string matches the glob pattern.
314    ///
315    /// # Arguments
316    ///
317    /// * `text` - The string to match against the pattern
318    ///
319    /// # Returns
320    ///
321    /// `true` if the string matches the pattern, `false` otherwise.
322    pub fn matches(&self, text: &str) -> bool {
323        // If we have expanded alternatives, check each of them
324        if !self.expanded_patterns.is_empty() {
325            return self.expanded_patterns.iter().any(|pat| {
326                GlobPattern::new(pat).matches(text)
327            });
328        }
329        
330        // Otherwise use the normal matching algorithm
331        self.matches_components(text, &self.components, 0)
332    }
333    
334    /// Match text against components starting from a position.
335    fn matches_components(&self, text: &str, components: &[Component], text_pos: usize) -> bool {
336        let text_chars: Vec<char> = text.chars().collect();
337        
338        self.matches_from_position(text, &text_chars, components, 0, text_pos)
339    }
340    
341    /// Recursive helper function to match text from a specific position.
342    fn matches_from_position(
343        &self, 
344        text: &str,
345        text_chars: &[char],
346        components: &[Component], 
347        component_idx: usize, 
348        text_pos: usize
349    ) -> bool {
350        // If we've reached the end of both the pattern and the text, it's a match
351        if component_idx >= components.len() {
352            return text_pos >= text_chars.len();
353        }
354        
355        // If we've reached the end of the text but not the pattern,
356        // it's only a match if the rest of the pattern can match empty string
357        if text_pos >= text_chars.len() {
358            // Special cases for components that can match empty strings
359            match &components[component_idx] {
360                Component::MultiWildcard => {
361                    return self.matches_from_position(text, text_chars, components, component_idx + 1, text_pos);
362                },
363                _ => {
364                    // Check if the rest of the pattern consists only of multi-wildcards
365                    for i in component_idx..components.len() {
366                        if !matches!(components[i], Component::MultiWildcard) {
367                            return false;
368                        }
369                    }
370                    return true;
371                }
372            }
373        }
374        
375        match &components[component_idx] {
376            Component::Literal(lit) => {
377                let lit_chars: Vec<char> = lit.chars().collect();
378                
379                if text_pos + lit_chars.len() > text_chars.len() {
380                    return false;
381                }
382                
383                for (i, &lit_char) in lit_chars.iter().enumerate() {
384                    if text_chars[text_pos + i] != lit_char {
385                        return false;
386                    }
387                }
388                
389                // Move past this literal in both pattern and text
390                self.matches_from_position(text, text_chars, components, component_idx + 1, text_pos + lit_chars.len())
391            },
392            Component::SingleWildcard => {
393                // ? matches exactly one character, so advance both
394                self.matches_from_position(text, text_chars, components, component_idx + 1, text_pos + 1)
395            },
396            Component::MultiWildcard => {
397                // * can match any number of characters (including zero)
398                
399                // Option 1: * matches nothing, move to next component
400                if self.matches_from_position(text, text_chars, components, component_idx + 1, text_pos) {
401                    return true;
402                }
403                
404                // Option 2: * matches the current character, try again at next position
405                self.matches_from_position(text, text_chars, components, component_idx, text_pos + 1)
406            },
407            Component::CharacterClass { chars, negated } => {
408                let matches_class = chars.contains(&text_chars[text_pos]) != *negated;
409
410                if matches_class {
411                    self.matches_from_position(text, text_chars, components, component_idx + 1, text_pos + 1)
412                } else {
413                    false
414                }
415            },
416            // Component::Alternatives(alternatives) => {
417            //     // This shouldn't be reached because we're expanding alternatives before matching
418            //     // But just in case, implement a simple matching
419            //     for alt in alternatives {
420            //         if GlobPattern::new(alt).matches(&text[text_pos..]) {
421            //             return true;
422            //         }
423            //     }
424            //     false
425            // }
426        }
427    }

429    /// Find all files in a directory that match the glob pattern.
430    ///
431    /// # Arguments
432    ///
433    /// * `base_dir` - The base directory to start searching from
434    ///
435    /// # Returns
436    ///
437    /// A vector of PathBuf instances that match the pattern.
438    pub fn find_files<P: AsRef<Path>>(&self, base_dir: P) -> std::io::Result<Vec<PathBuf>> {
439        let mut result = Vec::new();
440        
441        // If we have expanded alternatives, search with each pattern
442        if !self.expanded_patterns.is_empty() {
443            for pattern in &self.expanded_patterns {
444                let pattern_glob = GlobPattern::new(pattern);
445                let files = pattern_glob.find_files(base_dir.as_ref())?;
446                for file in files {
447                    if !result.contains(&file) {
448                        result.push(file);
449                    }
450                }
451            }
452            return Ok(result);
453        }
454        
455        // Otherwise use normal search
456        self.find_files_recursive(base_dir.as_ref(), &mut result)?;
457        Ok(result)
458    }
459
460    /// Recursively search for files matching the pattern.
461    fn find_files_recursive(&self, dir: &Path, results: &mut Vec<PathBuf>) -> std::io::Result<()> {
462        if !dir.is_dir() {
463            return Ok(());
464        }
465
466        for entry in fs::read_dir(dir)? {
467            let entry = entry?;
468            let path = entry.path();
469            
470            // Convert path to string for pattern matching
471            if let Some(path_str) = path.to_str() {
472                // Check if the path matches our pattern
473                if self.matches(path_str) {
474                    results.push(path.clone());
475                }
476            }
477            
478            // Recursively search directories
479            if path.is_dir() {
480                self.find_files_recursive(&path, results)?;
481            }
482        }
483
484        Ok(())
485    }
